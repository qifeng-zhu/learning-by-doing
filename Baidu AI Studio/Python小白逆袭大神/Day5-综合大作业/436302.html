def clear_special_char(content):
    '''
    
     content:
    return: 
    '''
    s = content.replace('\n', '').replace('\r', '')
    s = re.findall('[\u4e00-\u9fa5a-zA-Z0-9]+',s)
    s = ''.join(s)

def stopwordslist(file_path):
    '''
    
     file_path:
    returnlist
    '''
    with open(file_path + 'cn_stopwords.txt', 'r') as f:
        stopwords_list = f.readlines()
        stopwords_list = ''.join(stopwords_list).split('\n')
    return stopwords_list

def movestopwords(fenci_list, stopwords_list):
    '''
    ,
     file_path: stopwords:list counts: 
    returnNone
    '''
    normal_words = []
    for word in fenci_list:
