with open('data/data31557/20200422.json', 'r', encoding='UTF-8') as file:
         json_array = json.loads(file.read())

#,xy

zones = []
for star in json_array:
    zone = star['zone']
    zones.append(zone)
print(len(zones))
print(zones)


zone_list = []
count_list = []

for zone in zones:
    if zone not in zone_list:
        count = zones.count(zone)
        zone_list.append(zone)
        count_list.append(count)

print(zone_list)
print(count_list)

# 
plt.rcParams['font.sans-serif'] = ['SimHei'] #

plt.bar(range(len(count_list)), count_list,color='r',tick_label=zone_list,facecolor='#9999ff',edgecolor='white')

plt.yticks(fontsize=20)

import matplotlib.pyplot as plt
import numpy as np 
import json
import matplotlib.font_manager as font_manager
import pandas as pd
%matplotlib inline

df = pd.read_json('data/data31557/20200422.json')
#weight
df['weight_clean'] = df['weight'].str.extract('(\d+)kg').astype(int)

bin_edges = [0, 45, 50, 55, 100]
bin_names = ['<=45kg', '45~50kg', '50~55kg', '>55kg']
df['weight_bin'] = pd.cut(df['weight_clean'], bin_edges, labels = bin_names)
weight_count = df['weight_bin'].value_counts(ascending=True)

plt.figure(figsize=(8,8))
plt.rcParams['font.sans-serif'] = ['SimHei']
plt.title('2', fontsize = 24, pad = 40)
plt.pie(weight_count, labels = bin_names, startangle = 90, autopct = '%1.1f%%', explode = (0,0,0,0.05), textprops = {'fontsize': 16})
plt.axis('equal')
plt.savefig('/home/aistudio/work/result/pie_result03.jpg');
plt.show()
